Matikan tombol tab/kategori Fore Coffee di navigasi atas (kalau ada) document.querySelectorAll('button, a, .brand-category-pill, h2, h3').forEach(el => { if (el.innerText && el.innerText.trim() === "Fore Coffee" && !el.classList.contains('toko-tutup')) { el.classList.add('toko-tutup', 'opacity-50', 'pointer-events-none'); el.style.textDecoration = "line-through"; } }); // 2. Matikan Card Menu Fore Coffee const productCards = document.querySelectorAll('.modern-product-card, .featured-product-card, .product-card'); productCards.forEach(card => { const img = card.querySelector('img'); if (!img) return; // Kalau gak ada gambar, lewati // Cek apakah URL gambar di card ini ada di daftar gambar Fore const imgSrc = img.getAttribute('src'); const isClosed = gambarTutup.includes(imgSrc); if (isClosed && !card.classList.contains('toko-tutup')) { // Kasih class abu-abu dan matikan pointer card.classList.add('toko-tutup', 'relative', 'grayscale', 'opacity-80'); card.style.pointerEvents = "none"; // Bikin elemen stempel raksasa const closedOverlay = document.createElement('div'); closedOverlay.className = 'absolute inset-0 bg-black/60 z-50 flex items-center justify-center backdrop-blur-[2px] rounded-2xl pointer-events-auto cursor-not-allowed'; closedOverlay.innerHTML = `
CLOSED
`; // Tameng ekstra biar tombol add to cart di dalamnya 100% gak bisa dipencet closedOverlay.addEventListener('click', (e) => { e.preventDefault(); e.stopPropagation(); }, true); card.appendChild(closedOverlay); } }); } // Eksekusi langsung setTimeout(applyClosedState, 100); // Satpam penjaga: Pantau pergerakan web kalau pengunjung ganti-ganti tab kategori const observer = new MutationObserver(() => applyClosedState()); const appContainer = document.getElementById('app') || document.body; observer.observe(appContainer, { childList: true, subtree: true });
